328  def validate(request, match):
329  
330      #Get the respective match for P1 and P2 and check if exists
331      try:
332          game = Fixture.objects.get(id=match)
333      except Fixture.DoesNotExist:
334          messages.error(request, f'Não foi possível autenticar a partida #{match} dentro de sua tabela de jogos!')
335          return HttpResponseRedirect(request.META.get('HTTP_REFERER'))
336  
337      #A set of "players" for further check "in"
338      players = (player for player in game.players.all())
339  
340      #Get all the division enviroments
341      data = game.division.league.enviroment.retrieve()
342  
343      p1 = game.players.first()
344      p2 = game.players.last()
345  
346      # Checking if the match has a mirror in Result Table. If not, raise a erro validation.
347      try:
348          result_p1 = Result.objects.get(fixture=game, player=p1)
349      except Result.DoesNotExist:
350          messages.error(request, f'Não foi possível autenticar a partida #{match} dentro de sua tabela de jogos!')
351          return HttpResponseRedirect(request.META.get('HTTP_REFERER'))
352  
353      # Checking if the match has a mirror in Result Table. If not, raise a erro validation.
354      try:
355          result_p2 = Result.objects.get(fixture=game, player=p2)
356      except Result.DoesNotExist:
357          messages.error(request, f'Não foi possível autenticar a partida #{match} dentro de sua tabela de jogos!')
358          return HttpResponseRedirect(request.META.get('HTTP_REFERER'))
359  
360      if (request.user in players) and (request.user.username != game.submited_by):
361          game.status = 1
362          game.validation = 1
363          game.on_date = datetime.datetime.today()
364          result_p1.validation = 1
365          result_p1.enabled = True
366          result_p1.comment = 'Validado por '+request.user.username+'.'
367          result_p1.on_date = datetime.datetime.today()

369          result_p2.on_date = datetime.datetime.today()
370          result_p2.enabled = True
371          result_p2.comment = 'Validado por '+request.user.username+'.'
372  
373          # Update P1/P2 in Results regardless final status of the macth: WINNER, LOSER, DRAW.
374          if result_p1.sets == result_p2.sets:
375              p1.stat.bcmPoints = p1.stat.bcmPoints + 1
376              p1.stat.leaguePerformance = p1.stat.leaguePerformance[1:4] + 'D'
377              result_p1.final = 2
378              result_p1.points = game.division.league.enviroment.matchDrawPoints
379              p2.stat.bcmPoints = p1.stat.bcmPoints + 1
380              p2.stat.leaguePerformance = p2.stat.leaguePerformance[1:4] + 'D'
381              result_p2.final = 2
382              result_p2.points = game.division.league.enviroment.matchDrawPoints
383          else:
384              if result_p1.sets > result_p2.sets:
385                  p1.stat.bcmPoints = p1.stat.bcmPoints + data['meritpoints'][game.division.formation]
386                  p1.stat.leaguePerformance = p1.stat.leaguePerformance[1:4] + 'W'
387                  p1.stat.bcmWin = p1.stat.bcmWin + 1
388                  result_p1.final = 1
389                  result_p1.points = game.division.league.enviroment.matchWinPoints
390                  p2.stat.bcmPoints = p1.stat.bcmPoints + 0
391                  p2.stat.leaguePerformance = p2.stat.leaguePerformance[1:4] + 'L'
392                  result_p2.final = 0
393                  result_p2.points = 0
394              else:
395                  p1.stat.bcmPoints = p1.stat.bcmPoints + 0
396                  p1.stat.leaguePerformance = p1.stat.leaguePerformance[1:4] + 'L'
397                  result_p1.final = 0
398                  result_p1.points = 0
399                  p2.stat.bcmPoints = p1.stat.bcmPoints + data['meritpoints'][game.division.formation]
400                  p2.stat.leaguePerformance = p2.stat.leaguePerformance[1:4] + 'W'
401                  p2.stat.bcmWin = p2.stat.bcmWin + 1
402                  result_p2.final = 1
403                  result_p2.points = game.division.league.enviroment.matchWinPoints
404  
405          # Updating P1 Stat
406          if p1.stat.bcmAvg == 0:
407              p1.stat.bcmAvg = round(result_p1.average, 2)  # Rounding up 2 decimals values.
408          else:
409              p1.stat.bcmAvg = round(((p1.stat.bcmAvg + result_p1.average) / 2), 2)
410  
411          if p1.stat.best3da < p1.stat.bcmAvg:
412              p1.stat.best3da = round(p1.stat.bcmAvg, 2)
413  
414          p1.stat.bcmMatches = p1.stat.bcmMatches + 1
415          p1.stat.bcmTon = p1.stat.bcmTon + result_p1.ton
416          p1.stat.bcmTon40 = p1.stat.bcmTon40 + result_p1.ton40
417          p1.stat.bcmTon70 = p1.stat.bcmTon + result_p1.ton70
418          p1.stat.bcmTon80 = p1.stat.bcmTon80 + result_p1.ton80
419  
420          if p1.stat.bcmLeg == 0:
421              p1.stat.bcmLeg = result_p1.best_leg
422          else:
423              if p1.stat.bcmLeg > result_p1.best_leg:
424                  p1.stat.bcmLeg = result_p1.best_leg
425  
426          if p1.stat.bcmOut < result_p1.highest_out:
427              p1.stat.bcmOut = result_p1.highest_out
428  
429          p1.stat.leagueParticipation = p1.stat.leagueParticipation + 1
430  
431          # Updating P2 Stat
432          if p2.stat.bcmAvg == 0:
433              p2.stat.bcmAvg = round(result_p2.average, 2)
434          else:
435              p2.stat.bcmAvg = round(((p2.stat.bcmAvg + result_p2.average) / 2), 2)
436  
437          if p2.stat.best3da < p2.stat.bcmAvg:
438              p2.stat.best3da = round(p2.stat.bcmAvg, 2)

441          p2.stat.bcmTon = p2.stat.bcmTon + result_p2.ton
442          p2.stat.bcmTon40 = p2.stat.bcmTon40 + result_p2.ton40
443          p2.stat.bcmTon70 = p2.stat.bcmTon + result_p2.ton70
444          p2.stat.bcmTon80 = p2.stat.bcmTon80 + result_p2.ton80
445

447              p2.stat.bcmLeg = result_p2.best_leg
448          else:
449              if p2.stat.bcmLeg > result_p2.best_leg:
450                  p2.stat.bcmLeg = result_p2.best_leg
451

462  
463          messages.success(request, f'Sua partida #{match} foi validada com sucesso!')

465      else:
466          messages.error(request, f'Não foi possível autenticar a partida #{match} dentro de sua tabela de jogos!')
467          return HttpResponseRedirect(request.META.get('HTTP_REFERER'))

564  @login_required()
565  def ranking(request, slug):
566  
567      #Collecting all results for "division".
568      division = Division.objects.get(slug=slug)
569      results = Result.objects.filter(fixture__division=division)
570      finished = Fixture.objects.filter(division=division, validation=1)
571      pending = Fixture.objects.filter(division=division, validation=0)
572  
573      #Set the Q queries to build the ranking player data for division(slug).
574      members = Count('created_at')
575      matches = Count('enabled', filter=Q(validation=1))
576      points = Sum('points', filter=Q(enabled=True, validation=1))
577      difference = Sum('legs_diff', filter=Q(enabled=True, validation=1))
578      legs = Sum('legs',  filter=Q(enabled=True, validation=1))
579      wins = Count('final', filter=Q(final=1, enabled=True, validation=1))
580      losses = Count('final', filter=Q(final=0, enabled=True, validation=1))
581      draws = Count('final', filter=Q(final=2, enabled=True, validation=1))
582      avg = Avg('average', filter=Q(enabled=True, validation=1))
583      best = Min('best_leg', filter=Q(enabled=True, validation=1, best_leg__gt=0))
584      out = Max('highest_out', filter=Q(enabled=True, validation=1))
585      ton = Sum('ton', filter=Q(enabled=True, validation=1))
586      ton40 = Sum('ton40', filter=Q(enabled=True, validation=1))
587      ton70 = Sum('ton70', filter=Q(enabled=True, validation=1))
588      ton80 = Sum('ton80', filter=Q(enabled=True, validation=1))
589  
590      ranking = results.values('player').annotate(
591          matches=matches,
592          points=points,
593          difference=difference,
594          legs=legs,
595          wins=wins,
596          losses=losses,
597          draws=draws,
598          average=avg,
599          best=best,
600          out=out,
601          ton=ton,
602          ton40=ton40,
603          ton70=ton70,
604          ton80=ton80,
605          members=members
606      ).order_by('-points', '-difference', '-legs', '-wins', '-average')
607  
608      for line in range(len(ranking)):
609          u = User.objects.get(id=ranking[line]['player'])
610          ranking[line]['player'] = u
611  
612      database = list(ranking)
613  
614      response = {'database': database,
615                  'division': division,
616                  'finished': finished,
617                  'pending': pending}
618  
619      return render(request, 'user_ranking_show.html', response)
